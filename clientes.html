// ========================================================================= // IMPORTACIONES DE FIREBASE - MÓDULO DE CLIENTES // ========================================================================= import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js"; import { getFirestore, collection, getDocs, addDoc, doc, updateDoc, deleteDoc } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js"; // CONFIGURACIÓN DE TU PROYECTO (Asegúrate de que coincida con tus credenciales) const firebaseConfig = { apiKey: "TU_API_KEY", authDomain: "fumigadora-tecnoplagas.firebaseapp.com", projectId: "fumigadora-tecnoplagas", storageBucket: "fumigadora-tecnoplagas.appspot.com", messagingSenderId: "TU_SENDER_ID", appId: "TU_APP_ID" }; // Inicialización de Firebase y base de datos Firestore const app = initializeApp(firebaseConfig); const db = getFirestore(app); // Almacenamiento local temporal de los clientes descargados let clientesFirestore = {}; // ========================================================================= // 1. CARGA AUTOMÁTICA DE CLIENTES DESDE FIRESTORE EN LA TABLA O SELECT // ========================================================================= async function cargarClientes() { const selectCliente = document.getElementById('select-cliente'); const tablaClientesBody = document.getElementById('tabla-clientes-body'); try { console.log("⏳ Consultando la colección 'Clientes' en Firestore..."); const querySnapshot = await getDocs(collection(db, "Clientes")); // Si existe el select de clientes (ej: en la pantalla de emisión) if (selectCliente) { selectCliente.innerHTML = ''; } // Si existe la tabla (ej: en la pantalla de administración de clientes) if (tablaClientesBody) { tablaClientesBody.innerHTML = ''; } clientesFirestore = {}; querySnapshot.forEach((doc) => { const data = doc.data(); const idDocumento = doc.id; clientesFirestore[idDocumento] = data; // Opción A: Llenar el menú desplegable (Select) if (selectCliente) { const option = document.createElement('option'); option.value = idDocumento; // Soporta campos comunes: "Nombre Completo", "Nombre", "Razon Social" option.textContent = data["Nombre Completo"] || data["Nombre"] || data["Razón Social"] || idDocumento; selectCliente.appendChild(option); } // Opción B: Llenar la tabla de administración if (tablaClientesBody) { const tr = document.createElement('tr'); tr.innerHTML = ` ${data["Nombre Completo"] || data["Nombre"] || "---"} ${data["Cédula"] || data["Cedula"] || data["RUC"] || "---"} ${data["Teléfono"] || data["Telefono"] || "---"} ${data["Correo"] || data["Email"] || "---"} ${data["Dirección"] || data["Direccion"] || "---"} `; tablaClientesBody.appendChild(tr); } }); console.log("✅ Datos de la colección 'Clientes' cargados correctamente."); // Asignar eventos de editar/eliminar si estamos en la tabla de administración if (tablaClientesBody) { asignarEventosAcciones(); } } catch (error) { console.error("❌ Error al cargar los clientes desde Firestore: ", error); } } // ========================================================================= // 2. REGISTRAR UN NUEVO CLIENTE (FORMULARIO) // ========================================================================= const formularioCliente = document.getElementById('cliente-form'); if (formularioCliente) { formularioCliente.addEventListener('submit', async (e) => { e.preventDefault(); // Captura de datos respetando una estructura limpia y legible const nombreCompleto = document.getElementById('cliente-nombre').value; const cedula = document.getElementById('cliente-cedula').value; const telefono = document.getElementById('cliente-telefono').value; const correo = document.getElementById('cliente-correo').value; const direccion = document.getElementById('cliente-direccion').value; const idEdicion = document.getElementById('cliente-id-edicion') ? document.getElementById('cliente-id-edicion').value : ""; // Objeto estructurado para Firebase const datosCliente = { "Nombre Completo": nombreCompleto, "Cédula": cedula, "Teléfono": telefono, "Correo": correo, "Dirección": direccion, "fechaModificacion": new Date().toISOString() }; try { if (idEdicion) { // MODO EDICIÓN: Actualizar cliente existente console.log(`💾 Actualizando cliente con ID: ${idEdicion}`); const clienteRef = doc(db, "Clientes", idEdicion); await updateDoc(clienteRef, datosCliente); alert("✅ Cliente actualizado con éxito."); document.getElementById('cliente-id-edicion').value = ""; // Limpiar id de edición if (document.getElementById('btn-submit-cliente')) { document.getElementById('btn-submit-cliente').textContent = "Guardar Cliente"; } } else { // MODO CREACIÓN: Crear nuevo cliente console.log("💾 Registrando nuevo cliente en Firestore..."); await addDoc(collection(db, "Clientes"), datosCliente); alert("✅ Cliente registrado con éxito."); } formularioCliente.reset(); cargarClientes(); // Recargar datos } catch (error) { console.error("❌ Error al procesar el cliente en Firebase: ", error); alert("Hubo un error al guardar los datos del cliente."); } }); } // ========================================================================= // 3. ACCIONES DE LA TABLA (EDITAR / ELIMINAR) // ========================================================================= function asignarEventosAcciones() { // Eventos para Editar document.querySelectorAll('.btn-editar').forEach(boton => { boton.addEventListener('click', function() { const idDoc = this.getAttribute('data-id'); const cliente = clientesFirestore[idDoc]; if (cliente) { // Llenar el formulario con los datos para editar if (document.getElementById('cliente-nombre')) document.getElementById('cliente-nombre').value = cliente["Nombre Completo"] || ""; if (document.getElementById('cliente-cedula')) document.getElementById('cliente-cedula').value = cliente["Cédula"] || ""; if (document.getElementById('cliente-telefono')) document.getElementById('cliente-telefono').value = cliente["Teléfono"] || ""; if (document.getElementById('cliente-correo')) document.getElementById('cliente-correo').value = cliente["Correo"] || ""; if (document.getElementById('cliente-direccion')) document.getElementById('cliente-direccion').value = cliente["Dirección"] || ""; // Guardar el ID en un input oculto para saber que estamos editando if (document.getElementById('cliente-id-edicion')) { document.getElementById('cliente-id-edicion').value = idDoc; } if (document.getElementById('btn-submit-cliente')) { document.getElementById('btn-submit-cliente').textContent = "Actualizar Datos"; } // Desplazar la pantalla hacia el formulario formularioCliente.scrollIntoView({ behavior: 'smooth' }); } }); }); // Eventos para Eliminar document.querySelectorAll('.btn-eliminar').forEach(boton => { boton.addEventListener('click', async function() { const idDoc = this.getAttribute('data-id'); const nombre = clientesFirestore[idDoc] ? clientesFirestore[idDoc]["Nombre Completo"] : "este cliente"; if (confirm(`¿Estás completamente seguro de eliminar a ${nombre}?`)) { try { console.log(`🗑️ Eliminando documento: ${idDoc}`); await deleteDoc(doc(db, "Clientes", idDoc)); alert("🗑️ Cliente eliminado de la base de datos."); cargarClientes(); // Recargar listado } catch (error) { console.error("❌ Error al eliminar el cliente: ", error); } } }); }); } // ========================================================================= // 4. INICIALIZACIÓN AL CARGAR EL DOM // ========================================================================= document.addEventListener("DOMContentLoaded", () => { cargarClientes(); });
